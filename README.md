## aid-svg
>AiDesign svg图标库

# install

​```
npm install aid-svg
​```
# use

​```
// main.js
import SvgIcon from 'aid-svg'
Vue.use(SvgIcon)
​```

​```
// demo.js
<template>
  <svg-icon icon-class="alert-circle"></svg-icon>
</template>

<script>
export default {
  name: 'demo',
  data () {
    return {
    }
  }
}
</script>
